{

	int testcase;
	char s[1024];
	gets(s);
	sscanf(s,"%d",&testcase);
	for (int case_id=1;case_id<=testcase;case_id++)
	{
		gets(s);
		int c[26];
		memset(c,0,sizeof(c));
		for (int i=0;s[i];i++)
			if (islower(s[i]))
				c[s[i]-'a']++;
			else if (isupper(s[i]))
				c[s[i]-'A']++;
		sort(c,c+26);
		int R=0;
		for (int i=0;i<26;i++) R+=(i+1)*c[i];
		printf("Case #%d: %d\n",case_id,R);
	}
	return 0;
}
