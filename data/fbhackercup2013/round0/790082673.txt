#define LL long long
const double pi=acos(-1.0);
using namespace std;

int ti,ans,i,k,a[400];
char s[1000];

int cmp(int x,int y){
    return x>y;
}

int main(){
    scanf("%d",&ti);
    gets(s);
    for (k=1;k<=ti;k++){
        memset(a,0,sizeof(a));
        gets(s);
        for (i=0;s[i];i++) ++a[toupper(s[i])];
        sort(a+'A',a+'Z'+1,cmp);
        ans=0;
        for (i='A';i<='Z';i++) ans+=a[i]*(26-i+'A');
        printf("Case #%d: %d\n",k,ans);
    }
}
