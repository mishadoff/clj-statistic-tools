#include <cstdio>
#include <algorithm>
#define fi(a, b, c) for(int a = (b); a < (c); a++)
#define fd(a, b, c) for(int a = (b); a > (c); a--)
#define FI(a, b, c) for(int a = (b); a <= (c); a++)
#define FD(a, b, c) for(int a = (b); a >= (c); a--)
#define fe(a, b, c) for(int a = (b); a; a = c[a])
using namespace std;

int n, c[26], A;
char s[505];

int main(){
	freopen("beautiful_stringstxt.txt", "r", stdin);
	freopen("beautiful_stringstxt.out", "w", stdout);
	scanf("%d\n", &n);
	FI(z, 1, n){
		gets(s);
		A = 0;
		fi(i, 0, 26) c[i] = 0;
		for(int i = 0; s[i]; i++){
			if(s[i] >= 'a' && s[i] <= 'z') c[s[i] - 'a']++;
			if(s[i] >= 'A' && s[i] <= 'Z') c[s[i] - 'A']++;
		}
		sort(c, c + 26);
		FD(i, 25, 0) A += c[i] * (i + 1);
		printf("Case #%d: %d\n", z, A);
	}
	scanf("\n");
}
