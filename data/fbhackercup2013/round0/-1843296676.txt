#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
int cas;
char s[105];
int dp[105][105];

bool dfs(int st, int en) {
    if (dp[st][en] != -1) return dp[st][en];
    if (st >= en) return 1;
    if (en - st == 1) {
        if (s[st] <= 'z' && s[st] >= 'a') return 1;
        if (s[st] == ' ' || s[st] == ':') return 1;
        return 0;
    }
    if (en - st == 2) {
        if (s[st] == ':' && s[st + 1] == ')') return 1;
        if (s[st] == ':' && s[st + 1] == '(') return 1;
    }
    bool ret = 0;
    for (int i = st + 1; i < en; i++)
        ret |= dfs(st, i) && dfs(i, en);
    if (s[st] == '(' && s[en - 1] == ')') ret |= dfs(st + 1, en - 1);
    dp[st][en] = ret;
    return ret;
}

int main() {
    freopen("a.in", "r", stdin);
    freopen("a.out", "w", stdout);
    scanf("%d", &cas);
    getchar();
    for (int Cas = 1; Cas <= cas; Cas++) {
        gets(s);
        printf("Case #%d: ", Cas);
        memset(dp, -1, sizeof (dp));
        bool flag = dfs(0, strlen(s));
        if (flag) puts("YES");
        else puts("NO");
    }
    return 0;
}
