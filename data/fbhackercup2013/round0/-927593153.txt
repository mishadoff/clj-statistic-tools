#include <stdio.h>
#include <string.h>

using namespace std;
int deal(char *s){
	int mark[110][110] = {0};
	int len = strlen(s);
	for (int i = 0; i < len; i++) {
		char tmp = s[i];
		if (i == 0) {
			if (tmp == ')') 
				return 0;
			if (tmp == '(')
				mark[i][1] = 1;	
			else 
				mark[i][0] = 1;
		} else {
			if (tmp == ')') {
				if (s[i - 1] == ':') {
					for (int j = 0; j < 110; j++)
						mark[i][j] = mark[i - 1][j];
				} 
				for (int j = 1; j < 110; j++)
					if (mark[i - 1][j]) 
						mark[i][j - 1] = 1;
			} else if (tmp == '(') {
				if (s[i - 1] == ':') {
					for (int j = 0; j < 110; j++)
						mark[i][j] = mark[i - 1][j];
				} 
				for (int j = 0; j < 110; j++)
					if (mark[i - 1][j]) 
						mark[i][j + 1] = 1;
				
			} else {
				for (int j = 0; j < 110; j++)
					mark[i][j] = mark[i - 1][j];	
			}	
		}
	}
	return mark[len - 1][0];
}
int main(){
	freopen("balanced_smileystxt.txt", "r", stdin);
	freopen("out2.txt", "w", stdout);
	int m, cases = 0;
	scanf("%d", &m);
	getchar();
	char s[110];
	while (m--) {
		memset(s, 0, sizeof(s));
		gets(s);	
		printf("Case #%d: %s\n", ++cases, deal(s)?"YES":"NO");
	}
	return 0;
}
