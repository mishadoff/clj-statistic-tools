#include <cstdio>
#include <algorithm>
#define LL long long
#define fi(a, b, c) for(int a = (b); a < (c); a++)
#define fd(a, b, c) for(int a = (b); a > (c); a--)
#define FI(a, b, c) for(int a = (b); a <= (c); a++)
#define FD(a, b, c) for(int a = (b); a >= (c); a--)
#define fe(a, b, c) for(int a = (b); a; a = c[a])
using namespace std;

int t, n, k, m[200005], b, c, r, e[100005];

int main(){
	freopen("find_the_mintxt.txt", "r", stdin);
	freopen("find_the_mintxt.out", "w", stdout);
	scanf("%d", &t);
	FI(z, 1, t){
		scanf("%d %d %d %d %d %d", &n, &k, &m[0], &b, &c, &r);
		FI(i, 0, k) e[i] = 0;
		if(m[0] < k + 1) e[m[0]]++;
		fi(i, 1, k){
			m[i] = ((LL) m[i - 1] * b + c) % r;
			if(m[i] < k + 1) e[m[i]]++;
		}
		
		for(b = 0; e[b]; b++);
		e[b]++;
		m[k] = b;
		
		FI(i, k + 1, k + k){
			int t = m[i - k - 1];
			if(t < k + 1){
				e[t]--;
				if(t < b && !e[t]){
					m[i] = t;
					e[t]++;
					continue;
				}
			}
			while(e[b]) b++;
			e[b]++;
			m[i] = b;
		}
		
		n--;
		if(n <= k + k) c = n;
		else{
			k++;
			c = n - (n / k) * k;
			if(c < k - 1) c += k;
		}
		printf("Case #%d: %d\n", z, m[c]);
	}
	scanf("\n");
}
