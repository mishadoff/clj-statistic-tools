#include <string>
#include <iostream>
#include <memory.h>
#include <cstring>
using namespace std;
int visit[150][150];
int main()
{
	freopen("balanced_smileystxt.txt","r",stdin);
	freopen("bal.txt","w",stdout);
	char buf[512];
	gets(buf);
	int T;sscanf(buf,"%d",&T);
	for (int kase=1;kase<=T;++kase)
	{
		gets(buf);
		bool fail = false;
		if (buf[0]==')') fail = true;
		else
		{
			int n = strlen(buf);
			memset(visit,0,sizeof(visit));
			visit[0][buf[0]=='(']=1;
			for (int i=1;i<n;++i) for (int k=0;k<=n;++k)
			{
				int add = 0;
				if (buf[i]=='(') add=1;
				else if (buf[i]==')') add=-1;
				if (k-add>=0 && k-add<=n) visit[i][k]|=visit[i-1][k-add];
				if (add!=0 && buf[i-1]==':') visit[i][k]|=visit[i-1][k];
			}
			fail = (visit[n-1][0]!=1);
		}
		printf("Case #%d: %s\n",kase,fail?"NO":"YES");
	}
	return 0;
}
