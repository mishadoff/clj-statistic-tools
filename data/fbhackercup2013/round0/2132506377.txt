using namespace std;

int main()
{
    int m;
    cin >> m;
    cin.ignore();

    for(int k=1; k<=m; ++k){
        string s;
        getline(cin, s);
        int n = s.size();

        vector<vector<bool> > dp(n+1, vector<bool>(n+1, false));
        dp[0][0] = true;

        for(int i=0; i<n; ++i){
            for(int j=0; j<n; ++j){
                if(!dp[i][j])
                    continue;

                if(s[i] == ')'){
                    if(j > 0)
                        dp[i+1][j-1] = true;
                }else if(s[i] == '('){
                    dp[i+1][j+1] = true;
                }else if(s.substr(i, 2) == ":(" || s.substr(i, 2) == ":)"){
                    dp[i+2][j] = true;
                    dp[i+1][j] = true;
                }else{
                    dp[i+1][j] = true;
                }
            }
        }

        if(dp[n][0])
            cout << "Case #" << k << ": YES" << endl;
        else
            cout << "Case #" << k << ": NO" << endl;
    }

    return 0;
}
