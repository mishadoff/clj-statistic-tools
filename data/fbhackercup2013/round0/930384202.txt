int main()
{
    string s;
    getline(cin, s);
    int m = to_T<int>(s);
    for (int case_num = 1; case_num <= m; ++case_num)
    {
        getline(cin, s);

        int cnt[333] = {};
        rep(i, s.size())
        {
            char c = s[i];
            if (isalpha(c))
                ++cnt[tolower(c)];
        }

        vector<int> num;
        rep(i, 256)
            num.pb(cnt[i]);
        sort(all(num), greater<int>());

        ll res = 0;
        rep(i, 26)
            res += (26 - i) * num[i];
        printf("Case #%d: %lld\n", case_num, res);
    }
}
