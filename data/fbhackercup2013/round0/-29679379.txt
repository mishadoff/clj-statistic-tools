#include <iostream>
#include <algorithm>
using namespace std;


int solve(string s){
	int score[26] = {};
	int ans = 0;
	for(int i = 0 ; i < s.size() ; i++){
		if( 'A' <= s[i] && s[i] <= 'Z' ) score[s[i]-'A']++;
		if( 'a' <= s[i] && s[i] <= 'z' ) score[s[i]-32-'A']++;
	}
	sort(score,score+26);
	for(int i = 0 ; i < 26 ; i++)
		ans += score[i] * (i+1);
	return ans;
}

int main(){
	int n;
	cin >> n;
	string s;
	getline(cin,s);
	for(int i = 0 ; i < n ; i++){
		getline(cin,s);
		cout << "Case #" << i+1 << ": " << solve(s) << endl;
	}
}
