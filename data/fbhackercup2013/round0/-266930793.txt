#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
int cas;
int p[26];

int main() {
    freopen("a.in", "r", stdin);
    freopen("a.out", "w", stdout);
    cin >> cas;
    getchar();
    for (int Cas = 1; Cas <= cas; Cas++) {
        cout << "Case #" << Cas << ": ";
        string s;
        getline(cin, s);
        int ans = 0;
        memset(p, 0, sizeof (p));
        for (int i = 0; i < s.length(); i++) {
            char c = s[i];
            if (c > 'Z') c -= 32;
            if (c <= 'Z' && c >= 'A') p[c - 'A']++;
        }
        sort(p, p + 26);
        for (int i = 25; i >= 0; i--)
            ans += (i + 1) * p[i];
        cout << ans << endl;
    }
    return 0;
}
