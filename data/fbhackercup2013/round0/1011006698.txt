bool dp[111][111];

int main()
{
	freopen("input.txt","r",stdin);freopen("output.txt","w",stdout);
	int tests;
	scanf("%d\n",&tests);
	FOR(test,1,tests){
		string s;getline(cin,s);
		s=" "+s;
		memset(dp,false,sizeof(dp));
		dp[0][0]=true;
		for(int i=1;i<sz(s);++i)for(int deg=0;deg<=101;++deg){
			if(s[i]!=')' && s[i]!='('){
				dp[i][deg]|=dp[i-1][deg];
			}else
			if(s[i]=='('){
				if(deg)dp[i][deg]|=dp[i-1][deg-1];
				if(s[i-1]==':')dp[i][deg]|=dp[i-1][deg];
			}else
			if(s[i]==')'){
				dp[i][deg]|=dp[i-1][deg+1];
				if(s[i-1]==':')dp[i][deg]|=dp[i-1][deg];
			}
		}

		string ans = (dp[sz(s)-1][0])?"YES":"NO";
		cout<<"Case #"<<test<<": "<<ans<<endl;
	}
}
