#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;

int num[30];
char str[2000];

int main(){
   // freopen("in.txt","r",stdin);
   // freopen("out.txt","w",stdout);
    int t,cas=1;
    cin>>t;
    getchar();
    while(t--){
        gets(str);
        int len=strlen(str);
        memset(num,0,sizeof(num));
        for(int i=0;i<len;i++){
            if(str[i]>='a' && str[i]<='z') str[i]+=('A'-'a');
            if(str[i]>='A' && str[i]<='Z') num[str[i]-'A']++;
        }
        sort(num,num+26);
        int res=0;
        for(int i=0;i<26;i++) res+=num[i]*(i+1);
        printf("Case #%d: %d\n",cas++,res);
    }
    return 0;
}
