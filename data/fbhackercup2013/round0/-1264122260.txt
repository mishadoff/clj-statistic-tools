char str[1<<10];
int add[26];
int val(char c){
    if('a'<=c && c<='z')
        return add[c-'a']++;
    if('A'<=c && c<='Z')
        return add[c-'A']++;
    return 0;
}
int main()
{
    freopen("beautiful_stringstxt.txt","r",stdin);
    freopen("beautiful_strings_ans.txt","w",stdout);
    int m,ca=0;
    scanf("%d\n",&m);
    while(m--){
        gets(str);
        int ret = 0;
        mst(add,0);
        For(i,strlen(str))
            val(str[i]);
        sort(add,add+26);
        For(i,26)
            ret += (i+1)*add[i];
        printf("Case #%d: %d\n",++ca,ret);
    }
    return 0;
}
