using namespace std;

int T;
char s[505];
int cnt[27];

int main() {
  scanf("%d\n", &T);
  for (int it = 1; it <= T; it++) {
    gets(s);
    memset(cnt,0,sizeof(cnt));
    for (int i = 0; i < strlen(s); i++)
      if ('a' <= s[i] && s[i] <= 'z') cnt[s[i] - 'a' + 1]++;
      else if ('A' <= s[i] && s[i] <= 'Z') cnt[s[i] - 'A' + 1]++;
    sort(cnt + 1,cnt + 27);
    int ret = 0;
    for (int i = 26; i > 0; i--) ret += cnt[i] * i;
    printf("Case #%d: %d\n", it, ret);
  }
}
