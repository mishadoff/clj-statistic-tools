#include <iostream>
#include <fstream>
#include <string>
#include <cctype>
#include <cstring>
#include <algorithm>

void solve() {
  std::string s;
  int m, count[30], res;
  std::ifstream inf("input2.txt");
  std::ofstream ouf("output2.txt");
  inf >> m;
  std::getline(inf, s);
  for (int i = 0; i < m; ++i) {
    std::getline(inf, s);
    memset(count, 0, sizeof(count));
    for (int j = 0; j < (int)s.size(); ++j) {
      if (isalpha(s[j])) {
        ++count[toupper(s[j]) - 'A'];
      }
    }
    std::sort(count, count + 26);
    res = 0;
    for (int j = 0; j < 26; ++j) {
      res += count[j] * (j + 1);
    }
    ouf << "Case #" << i + 1 << ": " << res << "\n";
  }
  ouf.close();
}

int main() {
  solve();
}
