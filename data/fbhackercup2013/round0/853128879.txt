using namespace std;

int main()
{
    int m;
    cin >> m;
    cin.ignore();

    for(int k=1; k<=m; ++k){
        string s;
        getline(cin, s);
        int n = s.size();

        vector<int> x(26, 0);
        for(int i=0; i<n; ++i){
            if('A' <= s[i] && s[i] <= 'Z')
                ++ x[s[i]-'A'];
            else if('a' <= s[i] && s[i] <= 'z')
                ++ x[s[i]-'a'];
        }

        sort(x.begin(), x.end());
        int ret = 0;
        for(int i=0; i<26; ++i)
            ret += x[i] * (i+1);
        cout << "Case #" << k << ": " << ret << endl;
    }

    return 0;
}
