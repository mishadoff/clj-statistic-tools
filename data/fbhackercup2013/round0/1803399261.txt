#include <cstdio>
#include <assert.h>
#include <cstdlib>
#include <iostream>
#include <algorithm>
#include <vector>
#include <cmath>
#include <set>
#include <cstring>
using namespace std;
typedef long long ll;
const int L = 26;
int i, j, k, m, n, ans, f[L];
char ss[10002];

int main()
{
    freopen("fb.in", "r", stdin);
    freopen("fb.txt", "w", stdout);

    int ti, tn;
    cin >> tn;
    gets(ss);
    for (ti = 1; ti <= tn; ti++)
    {
        gets(ss);
        for (i = 0; i < L; i++) f[i] = 0;
        for (i = 0; ss[i]; i++)
            if (isalpha(ss[i]))
                f[tolower(ss[i])-'a']++;
        sort(f, f+L);
        ans = 0;
        for (i = 0; i < L; i++)
            ans += f[i] * (i + 1);

        cout << "Case #" << ti << ": " << ans << endl;
    }

    return 0;
}
