import string
n = int(raw_input())
for case in range(1,n+1):
  s = raw_input()
  ct = {}
  for c in s.lower():
    if c not in string.lowercase: continue
    if c in ct:
      ct[c]+=1
    else:
      ct[c]=1
  l = sorted(ct.items(), key = lambda x:-x[1])
  res = 0
  sc = 26
  for i in l:
    res += sc * i[1]
    sc-=1
  print "Case #%d: %d" % (case,res)
