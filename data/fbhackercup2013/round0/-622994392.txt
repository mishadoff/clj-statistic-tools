A = 26

def work(s):
	num_ch = [0] * A
	for ch in s:
		if ch.isalpha():
			num_ch[ord(ch.lower()) - ord('a')] += 1
	num_ch.sort()
	ret = 0
	for i in range(0, A):
		ret += num_ch[i] * (i + 1)
	return ret

tc = int(raw_input())
for cc in range(1, tc + 1):
	print("Case #%d: %d" % (cc, work(raw_input())))
