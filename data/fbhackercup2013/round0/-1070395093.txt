typedef long long int64;

bool check(char *s, int left, int pos)
{
    while (s[pos] != '\0') {
        if (s[pos] == '(') {
            bool flag = false;
            flag |= check(s, left + 1, pos + 1);
            if (pos > 0 && s[pos - 1] == ':')
                flag |= check(s, left, pos + 1);
            return flag;
        }
        if (s[pos] == ')') {
            bool flag = false;
            if (left > 0)
                flag |= check(s, left - 1, pos + 1);
            if (pos > 0 && s[pos - 1] == ':')
                flag |= check(s, left, pos + 1);
            return flag;
        }
        ++pos;
    }
    return (left ? false : true);
}

void solve_case()
{
    char s[110];
    gets(s);
    printf("%s\n", check(s, 0, 0) ? "YES" : "NO");
}
