#define inf (1<<29)
#define eps (1e-9)

#define flt(x,y) ((x)<(y)-eps)
#define fgt(x,y) flt(y,x)
#define fle(x,y) !fgt(x,y)
#define fge(x,y) !flt(x,y)

int main()
{
	int m;
	char dump[1000];
	scanf ("%d", &m);
	fgets (dump, 1000, stdin);
	for (int cc = 1; cc <=m; cc++) {
		fgets (dump, 1000, stdin);
		int l = strlen(dump);
		int f[26];
		memset(f, 0 ,sizeof(f));
		for (int i = 0; i < l; i++) {
			if (('A' <= dump[i] && dump[i] <='Z') || 
				('a' <= dump[i] && dump[i] <='z')) {
				f[tolower(dump[i])-'a']++;
			}
		}
		sort(f, f+26);
		int ans = 0;
		for (int i = 26; i >= 1; i--) {
			ans += i * f[i-1];
		}
		printf ("Case #%d: %d\n", cc, ans);
	}
	return 0;
}
