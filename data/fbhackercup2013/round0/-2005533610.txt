#include <algorithm>
#include <iostream>
#include <cassert>
#include <cstring>
#include <cstdio>
#include <vector>
#include <cmath>
#include <map>
#include <set>

using namespace std;

#define SIZE(A) ((int)A.size())
#define LENGTH(A) ((int)A.length())
#define MP(A,B) make_pair(A,B)
#define PB(A) push_back(A)

char s[20000];
int cnt[200];

int main()
{
	int t;
	scanf("%d%*c", &t);

	for (int tn=1; t--;)
	{
		cin.getline(s, 10000);
	 	memset(cnt, 0, sizeof(cnt));
	 	int res = 0;
	 	for (int i = 0; s[i]; i++)
	 		if (isalpha(s[i]))
		 		cnt[tolower(s[i])-'a']++;
		sort(cnt, cnt+26);
		for (int i = 25; i>=0; i--)
			res+=cnt[i]*(i+1);
	 	printf("Case %d: %d\n", tn++, res);
	}


	return 0;
}
