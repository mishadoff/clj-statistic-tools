#include <cstdio>
#include <cstring>

char s[1000];
char t[1000];
bool cancellable[1000];

// len, depth
bool dp[1000][1000];

bool get_ans(int n) {
	dp[0][0] = true;
	for (int i = 1; i <= n; i++) {
		dp[0][i] = false;
	}
	for (int l = 1; l <= n; l++) {
		for (int j = 0; j <= n; j++) {
			if (t[l-1] == '(') {
				dp[l][j] = (cancellable[l-1] && dp[l-1][j]) || (j > 0 && dp[l-1][j-1]);
			} else {
				dp[l][j] = (cancellable[l-1] && dp[l-1][j]) || (j < n && dp[l-1][j+1]);
			}
		}
	}
	return dp[n][0];
}

int main() {
	int ncases;
	scanf("%d", &ncases);
	char dc;
	do {
		dc = fgetc(stdin);
	} while (dc != '\n');
	for (int cn = 1; cn <= ncases; cn++) {
		fgets(s, 1000, stdin);
		int sl = strlen(s);
		int l = 0;
		for (int i = 0; i < sl; i++) {
			if (s[i] == '(' || s[i] == ')') {
				t[l] = s[i];
				cancellable[l] = (i > 0 && s[i-1] == ':');
				l++;
			}
		}
		printf("Case #%d: %s\n", cn, get_ans(l) ? "YES" : "NO");
	}
}
