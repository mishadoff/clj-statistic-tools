#include <stdio.h>
#include <string.h>

#define MAXS (128)

int T;
char S[MAXS];

int dp[MAXS][MAXS];

int rec(int l, int r)
{
	if(l > r)
		return 1;

	int res = dp[l][r];
	if(res == -1)
	{
		if(l == r) // length 1
		{
			if(S[l] == '(' || S[l] == ')')
				res = 0;
			else
				res = 1;
		}
		else if(l + 1 == r) // length 2
		{
			if(S[l] == ':' || (S[l] == '(' && S[r] == ')'))
				res = 1;
			else if(S[l] == '(' || S[l] == ')' || S[r] == '(' || S[r] == ')')
				res = 0;
			else
				res = 1;
		}
		else // more
		{
			res = 0;

			if(S[l] == '(' && S[r] == ')')
			{
				if(rec(l + 1, r - 1))
					res = 1;
			}

			if(res == 0)
			{
				int i;
				for(i = l; i < r; ++i)
				{
					if(rec(l, i) && rec(i + 1, r))
					{
						res = 1;
						break;
					}
				}
			}
		}

		dp[l][r] = res;
	}

	return res;
}

int main()
{
	int i;

	scanf("%d", &T);
	for(i = 0; i < T; ++i)
	{
		scanf("\n%[^\n]", S);

		int l = strlen(S);
		memset(dp, -1, sizeof(dp));

		printf("Case #%d: ", i + 1);

		if(rec(0, l - 1))
			printf("YES\n");
		else
			printf("NO\n");
	}

	return 0;
}
