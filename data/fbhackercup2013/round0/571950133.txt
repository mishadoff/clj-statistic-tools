const char* ans[2] = {"NO", "YES"};

char s[1000];
bool f[2000][2000];

int main()
{
    #ifdef __FIO
    freopen("in.txt", "r", stdin);
    freopen("out.txt", "w", stdout);
    #endif
    int T, i0;
    scanf("%d", &T);
    gets(s);
    for(i0 = 1; i0 <= T; i0++){
        int n;
        int i, j;
        gets(s);
        n = strlen(s);
        memset(f, 0, sizeof f);
        f[0][0] = 1;
        for(i = 0; i < n; i++)
            for(j = 0; j < n; j++){
                if(f[i][j]){
                    if(s[i] != '(' && s[i] != ')')
                        f[i+1][j] = 1;
                    if(s[i] == '(')
                        f[i+1][j+1] = 1;
                    if(s[i] == ')' && j > 0)
                        f[i+1][j-1] = 1;
                    if(s[i] == ':' && (s[i+1] == ')' || s[i+1] == '('))
                        f[i+2][j] = 1;
                }
            }
        printf("Case #%d: %s\n", i0, ans[f[n][0]]);
    }
    return 0;
}
