	String l;
	boolean go(int i, int pts){
		if(i==l.length()){
			return pts==0;
		}	
		char c = l.charAt(i);
		boolean r = false;
		if(c==':'){
			if(i<l.length()-1){
				char cn=l.charAt(i+1);
				if(cn=='(' || cn==')'){
					r = r || go(i+2,pts);
				}
			}
			r = r || go(i+1,pts);
		}
		else if(c=='('){
//			print(c);
			r = r || go(i+1,pts>=0?pts+1:pts);
		}else if(c==')'){
			r = r || go(i+1,pts-1);
		}else{
			r = r || go(i+1,pts);
		}
		return r;
	}
