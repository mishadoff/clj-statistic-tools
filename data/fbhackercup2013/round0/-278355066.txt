#include <iostream>
using namespace std;
char input[1000];
int cts[26];
int main() {
    freopen("beautiful.in","r",stdin);
    freopen("beautiful.out","w",stdout);
    int tt = 1;
    gets(input); int T = atoi(input); while (T--) {
        printf("Case #%d: ",tt++);
        gets(input);
        memset(cts,0,26*sizeof(int));
        for (int i=0; input[i]; i++) {
            if (input[i]>='a' && input[i]<='z') cts[input[i]-'a']++;
            else if (input[i]>='A' && input[i]<='Z') cts[input[i]-'A']++;
        }
        sort(cts,cts+26);
        int tot = 0;
        for (int i=0; i<26; i++) {
            tot += (i+1)*cts[i];
        }
        printf("%d\n",tot);
    }
}
