int get_score( char c ){ 
    if( c >='A' && c <='Z' )  c = c-'A'+'a';  
    if( c >='a' && c <='z' ) return c-'a';  
    return -1; 
}
int main (){ 
    freopen("a.in","r",stdin ) ; 
    freopen("a.out","w",stdout ) ;  

    int  m;  
    scanf ("%d\n", &m) ;  

    for(int Cas = 1 ; Cas <= m ; ++Cas ){ 
        char ss[50000] ;
        gets(ss) ;
        vector< int > vc(26); 
        for(int i= 0 ; i < 26 ; ++i) vc[i]=0; 
        for(int i=0; ss[i]; ++i){ 
            int x = get_score( ss[i] ) ; 
            if( x == -1) continue ;   
            vc[ x ]++ ; 
        }
        sort(vc.begin(),vc.end()); 
        reverse(vc.begin(),vc.end()) ; 
        int ans = 0  ; 
        for( int i=0 ; i < 26 ; ++i){ 
            ans = ans + vc[i]*(26-i) ; 
        }
        cout << "Case #"<<Cas<<": "<<ans << endl; 
    }
    return 0 ; 
}
