#include<cstdio>
#include<algorithm>
#include<cstring>
using namespace std;

int n;
char str[1000];
int A[1000];

int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	scanf("%d",&n);
	gets(str);
	for(int c=1;c<=n;c++)
	{
		for(int d=0;d<1000;d++) A[d] = 0;
		gets(str);
		int k = strlen( str );
		for(int d=0;d<k;d++)
		{
			if( str[d] >= 'a' ) A[ str[d] - 'a' ] ++;
			else A[ str[d] - 'A' ] ++;
		}
		sort( A , A+1000 );
		int ans = 0;
		for(int d=0;d<26;d++)
		{
			int r = 26 - d;
			ans += A[ 999-d ] * r;
		}
		printf("Case #%d: %d\n",c,ans);
	}
}
