char s[105];
int T;
bool f[105][105];

bool checkOK() {
  memset(f,false,sizeof(f));
  f[0][0] = true;
  int L = strlen(s);
  for (int i = 1; i <= L; i++)
    for (int j = 0; j <= i; j++)    
      if (s[i - 1] == '(') {
        if (j > 0) f[i][j] |= f[i - 1][j - 1];
	if (i > 1 && s[i - 2] == ':') f[i][j] |= f[i - 1][j];
      }
      else if (s[i - 1] == ')') {
        f[i][j] |= f[i - 1][j + 1];
	if (i > 1 && s[i - 2] == ':') f[i][j] |= f[i - 1][j];
      }
      else f[i][j] |= f[i - 1][j];
  return f[L][0];
}

int main() {
  scanf("%d\n", &T);
  for (int it = 1; it <= T; it++) {
    gets(s);
    printf("Case #%d: ", it);
    if (checkOK()) printf("YES\n"); else printf("NO\n");
  }
}
