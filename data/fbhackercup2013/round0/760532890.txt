#define For(i,n) for(int i=0;i<n;i++)
#define sz(i) int(i.size())
#define mst(i,n) memset(i,n,sizeof(i))
#define eps 1e-11
#define MOD 1000000007
#define LL long long
#define pi acos(-1)
#define ALL(n) n.begin(),n.end()
#define pb push_back
#define mp make_pair
#define iFor(i,n) for(typeof(n.begin()) i=n.begin();i!=n.end();i++)
#define jFor(i,j,n) for(int i=j;i<n;i++)

int n,k;
int a,b,c,r;
int m[300005];

map<int,int>ct;

int main()
{
    freopen("find_the_mintxt.txt","r",stdin);
    freopen("find_the_mintxt.txt_ans.txt","w",stdout);
    int t,ca=0;
    scanf("%d\n",&t);
    while(t--){
        scanf("%d %d",&n,&k);
        scanf("%d %d %d %d",&a,&b,&c,&r);
        ct.clear();
        For(i,k){
            if(!i) m[i]=a;
            else m[i]=(1LL*m[i-1]*b+c)%r;
            ct[ m[i] ] = 0;
            //cout<<m[i]<<" ";
        }
        //cout<<endl;
        For(i,k+1) ct[i]=0;
        For(i,k) ct[ m[i] ]++;
        int mx = 0;
        jFor(i,k,k*3+2){
            if(i>k && m[i-k-1] <= mx && ct[ m[i-k-1] ]==0){
                m[i] = m[i-k-1];
            }
            else{
                while(ct[mx]>0) mx++;
                m[i] = mx;
            }
            ct[ m[i] ]++;
            ct[ m[i-k] ]--;
            //cout<<m[i]<<" ";
            if(i==k*2) mx = 0;
            //if(i==k*2) cout<<endl;
        }
        printf("Case #%d: ",++ca);
        n--;
        if(n<k*2+1) printf("%d\n",m[n]);
        else printf("%d\n",m[ ((n-k)%(k+1)) + 2*k+1 ]);
    }
    return 0;
}
