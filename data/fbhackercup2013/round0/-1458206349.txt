int cnt[222];

int main()
{
	//freopen("input.txt","r",stdin);freopen("output.txt","w",stdout);
	int tests;
	scanf("%d\n",&tests);
	FOR(test,1,tests){
		string s;getline(cin,s);
		for(int i=0;i<sz(s);++i)if(s[i]>='a' && s[i]<='z')s[i]-='a'-'A';
		memset(cnt,0,sizeof(cnt));
		vector<int >g;g.clear();
		for(int i=0;i<sz(s);++i)if(s[i]>='A' && s[i]<='Z')cnt[s[i]]++;
		FOR(i,'A','Z')g.pb(cnt[i]);sort(g.begin(),g.end());
		int ans=0;
		FOR(i,0,sz(g)-1)ans+=g[i]*(i+1);
		cout<<"Case #"<<test<<": "<<ans<<endl;
	}
}
