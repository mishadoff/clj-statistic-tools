import string
n = input()
for t in range(n):
    s = raw_input().lower()
    m = {c: 0 for c in string.lowercase}
    for c in s:
        if c in string.lowercase:
            m[c] += 1
    m = list(reversed(sorted([m[c] for c in m])))
    score = 0
    for i in range(len(m)):
        score += m[i] * (26-i)
    print "Case #%d: %d"%(t+1, score)
