#include<stdio.h>
#include<string.h>

int T, L;
char S[128];
int DP[128][128];

int main() {
	fgets(S, 100, stdin);
	sscanf(S, "%d", &T);
	for(int cs = 1; cs <= T; ++cs) {
		fgets(S, 120, stdin);
		L = strlen(S);
		memset(DP, 0, sizeof(DP));
		DP[0][0] = 1;
		for(int i = 0; i < L; ++i) {
			for(int j = 0; j < L; ++j) if(DP[i][j]) {
				if(S[i] == '(') DP[i+1][j+1] = 1;
				else if(S[i] == ')') {
					if(j > 0) DP[i+1][j-1] = 1;
				} else {
					DP[i+1][j] = 1;
					if(S[i] == ':' && i+1 < L && (S[i+1] == '(' || S[i+1] == ')')) {
						DP[i+2][j] = 1;
					}
				}
			}
		}
		printf("Case #%d: ", cs);
		puts(DP[L][0]?"YES":"NO");
	}
}
