int main() {
	int T;
	cin>>T;
	cin.ignore();
	REP(ttt, T) {
		string s;
		getline(cin, s);
		
		VI hi(26);
		REP(i, s.size()) {
			s[i] = tolower(s[i]);
			if('a' <= s[i] && s[i] <= 'z') hi[s[i]-'a']++;
		}
		sort(ALL(hi));
		ll ans=0;
		REP(i, 26) ans += (i+1) * hi[i];
		
		cout<<"Case #"<<ttt+1<<": "<<ans<<endl;
	}
	return 0;
