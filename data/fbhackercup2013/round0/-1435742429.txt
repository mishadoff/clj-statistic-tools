#include<iostream>
#include<algorithm>
#include<vector>
#include<cctype>
using namespace std;

int main() {
	int n;
	cin>>n;
	string a;
	getline(cin,a);
	for(int j=0;j<n;j++) {
		getline(cin,a);
		vector<int> count(26,0);
		for(int i=0;i<a.size();i++) {
			if(isalpha(a[i]))
				count[tolower(a[i])-'a']++;
		}
		sort(count.begin(),count.end());
		int score = 0;
		for(int i=25;i>=0;i--)
			score+=(i+1)*count[i];
		cout<<"Case #"<<j+1<<": "<<score<<endl;


	}

}
