#include <iostream>
#include <vector>
#include <algorithm>
#include <cstring>
#include <map>
#include <set>
#include <queue>

using namespace std;

int main() {
  int N; cin >> N;
  string ln;
  getline(cin, ln);
  for(int i = 0; i < N; i++) {
    getline(cin, ln);
    map<char, int> mp;
    for(int j = 0; j < ln.size(); j++) {
      if(isalpha(ln[j])) {
        mp[tolower(ln[j])]++;
      }
    }
    vector<int> V;
    for(typeof(mp.begin()) it = mp.begin(); it != mp.end(); it++) {
      V.push_back(it->second);
    }
    sort(V.rbegin(), V.rend());

    int r = 0;
    for(int j = 0; j < V.size(); j++) {
      r += (26 - j) * V[j];
    }
    cout << "Case #" << (i + 1) << ": " << r << endl;
  }
  return 0;
}
