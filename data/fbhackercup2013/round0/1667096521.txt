#include <iostream>
#include <sstream>
using namespace std;

bool solve(string s) {
    int depth_max = 0, depth_min = 0;
    for (int i = 0; i < s.size(); i++) {
        if (s[i] == '(') {
            depth_max++;
            depth_min++;
        }
        else if (s[i] == ')') {
            depth_max--;
            depth_min--;
        }
        else if (i < s.size()-1 && s[i] == ':' && s[i+1] == ')') {
            depth_min--;
            i++;
        }
        else if (i < s.size()-1 && s[i] == ':' && s[i+1] == '(') {
            depth_max++;
            i++;
        }
        if (depth_max < 0) return false;
    }
    return depth_min <= 0 && depth_max >= 0;
}

int main() {
    string line;
    getline(cin, line);
    stringstream ss(line);
    int T; ss >> T;
    for (int t = 1; t <= T; t++) {
        getline(cin, line);
        cout << "Case #" << t << ": " << (solve(line) ? "YES" : "NO") << endl;
    }
}
