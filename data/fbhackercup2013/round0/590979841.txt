ALPHA = 26

def most_beauty(s):
    freq = [0 for x in range(ALPHA)]
    for c in s:
        if c.islower():
            freq[ord(c)-ord('a')] += 1
    freq.sort()
    ans = 0
    for i, f in enumerate(freq):
        ans += f*(i+1)
    return ans

n = int(input())

for i in range(n):
    print("Case #{}:".format(i+1), most_beauty(input().lower()))
