#define PB(A) push_back(A)

int n, col;
char s[200];
int w[200][200];

int dfs(int u, int k)
{
	if (u == n) return !k;
 	if (w[u][k] == col) return 0;
 	w[u][k] = col;

	if (s[u]!='(' && s[u]!=')') return dfs(u+1, k);

	if (u && s[u-1]==':') if (dfs(u+1, k)) return 1;

//		printf("Here u=%d k=%d\n", u, k);

	if (s[u] == '(') return dfs(u+1, k+1);
	if (!k) return 0;

 	return dfs(u+1, k-1);
}

int main()
{
	int t;
	cin >> t;scanf("%*c");
	for (int tn=1; t--;)
	{
	 	col++;
	 	n = 0;
	 	for (int ch = getchar(); ch!='\n'; ch = getchar())
	 		s[n++] = ch;
	 	printf("Case #%d: ", tn++);
	 	if (dfs(0, 0))
	 		puts("YES");
	 	else
	 		puts("NO");
	}

	return 0;
}
