namespace BeautifulStrings {
char str[555];
int cnt[27];

void solve(int test_num) {
  memset(cnt, 0, sizeof (cnt));
  cin.getline(str, 510);
  int len = strlen(str);
  for (int i = 0; i < len; i++)
    if (isalpha(str[i]))
      cnt[tolower(str[i]) - 'a']++;
  int ans = 0;
  sort(cnt, cnt + 26);
  for (int i = 0; i < 26; i++)
    ans += cnt[i] * (i + 1);
  printf("Case #%d: %d\n", test_num, ans);
}

void solve() {
  #ifdef AZN
  freopen("input.txt", "r", stdin);
  freopen("output.txt", "w", stdout);
  freopen("azn.txt", "w", stderr);
  #endif
  int tests;
  scanf("%d ", &tests);
  for (int i = 1; i <= tests; i++)
    solve(i);
}
}

int main() {
  BeautifulStrings::solve();
  return 0;
