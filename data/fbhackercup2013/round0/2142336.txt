#include <iostream>
#include <vector>
using namespace std;
int main(){
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);
	int t;
	char tmp_str[200];
	cin >> t;
	cin.getline(tmp_str, 200);
	
	vector<vector<int>> a(100, 100);
	int n;
	for(int tt = 0; tt < t; tt++) {
		cin.getline(tmp_str, 600);
		string s = tmp_str;
		n = s.length();
		for(int i = 0; i < n; i++) {
			for(int j = 0; j < n; j++) {
				if (j < i) 
					a[i][j] = 1;
				else 
					a[i][j] = 0;
			}
		}
		for(int i = 0; i < n; i++) {
			if (s[i] == ':' || s[i] == ' ' || (s[i] >= 'a' && s[i] <= 'z'))
				a[i][i] = 1;
		}
		int res;
		for(int jj = 1; jj < n; jj++) {
			for(int i = 0; i < n - jj; i++) {
				res = 0;
				int j = i + jj;
				a[i][j] = 0;
				if (s[i] == '(')
					for(int k = j; k > i; k--) {
						if (s[k] == ')') {
							res = 1;
							if (i + 1 <= k - 1)
								res = res && a[i+1][k-1];
							if (k + 1 <= j)
								res = res && a[k+1][j];
							if (res == 1){
								a[i][j] = 1;
								break;
							}
						}
					}
				else if (s[i] == ':') {
					if (i + 1 <= j)
						a[i][j] = a[i + 1][j];
					else a[i][j] = 1;

					if (a[i][j] == 0 && (s[i + 1] == '(' || s[i + 1] == ')')) {
						if (i + 2 <= j)
							a[i][j] = a[i+2][j];
						else
							a[i][j] = 1;
					}
				}
				else if (s[i] == ')') a[i][j] = 0;
				else if (s[i] >= 'a' && s[i] <= 'z' || s[i] == ' ') {
					if (i + 1 <= j)
						a[i][j] = a[i+1][j];
					else 
						a[i][j] = 1;
				}
			}
		}
		cout<<"Case #"<<tt+1<<": ";
		if (a[0][n-1] == 1) 
			cout<<"YES";
		else
			cout<<"NO";
		cout<<endl;
	}
	return 0;
}
