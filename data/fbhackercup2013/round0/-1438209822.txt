int vrat(char c) {
    if(c>='a' && c<='z') return c-'a';
    if(c>='A' && c<='Z') return c-'A';
    return -1;
}

int main()
{
    int t;
    scanf("%d ",&t);
    For(i1,t) {
        string s;
        getline(cin,s);
        vector<int> P; P.resize(26,0);
        For(i,s.length()) {
            int x=vrat(s[i]);
            if(x!=-1) P[x]++;
        }
        sort(P.begin(),P.end());
        reverse(P.begin(),P.end());
        int vys=0,p=26;
        For(i,P.size()) {vys+=p*P[i]; p--;}
        printf("Case #%d: %d\n",i1+1,vys);
    }
return 0;
}
