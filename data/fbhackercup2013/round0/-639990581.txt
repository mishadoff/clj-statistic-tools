#include <cstdio>
#include <cstring>
#include <cstdlib>

int t,id;
int i,a[27],ans,len;
char st[1000];

int cmp(const void *a,const void *b)
{return *(int *)a- *(int *)b;}

int main()
{
	freopen("beautiful_stringstxt.txt","r",stdin);freopen("a.out","w",stdout);
	scanf("%d",&t);getchar();
	for (id=1;id<=t;id++)
	{
		memset(a,0,sizeof(a));
		gets(st);
		len=strlen(st);
		for (i=0;i<len;i++)
		{
			if (st[i]>='a' && st[i]<='z')a[st[i]-'a']++;
			if (st[i]>='A' && st[i]<='Z')a[st[i]-'A']++;
		}
		qsort(a,27,sizeof(a[0]),cmp);
		ans=0;
		for (i=26;i>0;i--)ans+=a[i]*i;
		printf("Case #%d: %d\n",id,ans);
	}
	return 0;
}
