#include <cstdio>

void solve(const char* s) {
  bool dp[105][105] = {{}};
  dp[0][0] = true;

  int i;
  for (i = 0; s[i] && s[i] != '\n'; ++i) {
    for (int b = 0; b < 105; ++b) {
      if (!dp[i][b]) continue;

      if (s[i] != '(' && s[i] != ')' && s[i] != ':') {
        dp[i+1][b] = true;
      } else {
        if (s[i] == ':' && s[i+1] == '(') {
          dp[i+2][b] = true;
        } else if (s[i] == ':' && s[i+1] == ')') {
          dp[i+2][b] = true;
        }
        
        if (s[i] == ':') {
          dp[i+1][b] = true;
        }
        
        if (s[i] == '(') {
          dp[i+1][b+1] = true;
        }
        
        if (s[i] == ')') {
          if (b > 0) {
            dp[i+1][b-1] = true;
          }
        }
      }
    }
  }

  if (dp[i][0])
    printf("YES\n");
  else
    printf("NO\n");
}

int main() {
  int n;
  scanf("%d\n", &n);
  for (int i = 0; i < n; ++i) {
    char buf[105] = {};
    fgets(buf, 105, stdin);
    printf("Case #%d: ", i+1);
    solve(buf);
  }

  return 0;
}
