#include<stdio.h>
#include<string.h>
#include<ctype.h>
#include<algorithm>
using namespace std;
char s[555];
int w[26];
int main(){
	int _;
	gets(s);
	sscanf(s,"%d",&_);
	for(int t=1; t<=_; t++){
		gets(s);
		int res=0;
		memset(w,0,sizeof(w));
		for(int i=0; s[i]; i++)
			if(isalpha(s[i]))
				w[tolower(s[i])-'a']++;
		sort(w,w+26);
		for(int i=0; i<26; i++)
			res+=w[i]*(i+1);
		printf("Case #%d: %d\n",t,res);
	}
	return 0;
}
