typedef long long int64;

void solve_case()
{
    char s[510];
    int c[26], sum = 0;
    memset(c, 0, sizeof(c));
    gets(s);
    for (int i = 0; s[i] != '\0'; ++i) {
        if (s[i] >= 'a' && s[i] <= 'z')
            c[s[i] - 'a']++;
        else if (s[i] >= 'A' && s[i] <= 'Z')
            c[s[i] - 'A']++;
    }
    sort(c, c + 26);
    for (int i = 0; i < 26; ++i)
        sum += c[i] * (i + 1);
    printf("%d\n", sum);
}

        printf("Case #%d: ", icase);
        solve_case();
