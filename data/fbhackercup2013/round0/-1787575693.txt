void run(){
  string s;
  int tsts; cin>>tsts; getline(cin,s);
  for (int tst=1; tst<=tsts; ++tst){
    ll hits[32]={};
    getline(cin,s);
    for (char c: s) if (c>='a' and c<='z') ++hits[c-'a']; else if (c>='A' and c<='Z') ++hits[c-'A'];
    sort(hits,hits+26);
    ll res=0;
    for (int i=0; i<26; i++){
      res+=(i+1)*hits[i];
    }
    cout<<"Case #"<<tst<<": "<<res<<endl;
  }
}
