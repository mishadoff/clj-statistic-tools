#include <cstdio>
#include <cstring>
#include <algorithm>

int m, cnt[30], ans;
char s[600];

int main() {

	freopen("pa.in", "r", stdin);
	freopen("pa.out", "w", stdout);

	scanf("%d\n", &m);
	for(int tc=1; tc<=m; tc++) {
		ans = 0;
		memset(cnt, 0, sizeof(cnt));
		fgets(s, 600, stdin);
		for(int i=0; s[i]; i++) {
			if(s[i] >= 'A' && s[i] <= 'Z') {
				cnt[ s[i]-'A' ] ++;
			}
			else if(s[i] >= 'a' && s[i] <= 'z') {
				cnt[ s[i]-'a' ] ++;
			}
		}
		std::sort(cnt, cnt+26);
		for(int i=0; i<26; i++)
			ans += cnt[i]*(i+1);
		printf("Case #%d: %d\n", tc, ans);
	}
}
