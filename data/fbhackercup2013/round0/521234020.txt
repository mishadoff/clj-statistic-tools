#!/usr/bin/env python

import string

def main():
	n = int( raw_input() )
	for x in range(n):
		s = raw_input()
		count = [0] * 26
		for c in s:
			if c in string.ascii_uppercase:
				count[ ord(c) - ord('A')] += 1
			elif c in string.ascii_lowercase:
				count[ ord(c) - ord('a')] += 1
		count.sort()
		sum = 0
		for i in range(26):
			sum += (i+1) * count[i]
		print 'Case #%d: %d' % ( x+1,sum ) 


if __name__ == '__main__':
	main()
