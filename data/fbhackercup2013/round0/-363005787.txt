#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

char s[1000];
int counts[26];

int main() {
	int m;
	scanf("%d", &m);
	char nl;
	do {
		nl = fgetc(stdin);
	} while (nl != '\n');
	for (int i = 1; i <= m; i++) {
		fgets(s, 1000, stdin);
		int len = strlen(s);
		memset(counts, 0, sizeof(int) * 26);
		for (int j = 0; j < len; j++) {
			if (s[j] >= 'a' && s[j] <= 'z') {
				counts[s[j] - 'a']++;
			} else if (s[j] >= 'A' && s[j] <= 'Z') {
				counts[s[j] - 'A']++;
			}
		}
		sort(counts, counts + 26);
		int ans = 0;
		for (int j = 0; j < 26; j++) {
			ans += (j + 1) * counts[j];
		}
		printf("Case #%d: %d\n", i, ans);
	}
}
