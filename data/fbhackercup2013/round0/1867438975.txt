import string
letters = string.ascii_lowercase
scores = range(1, 27)[::-1]

def solve(s):
  s = filter(lambda x: x in letters, s.strip().lower())
  counts = dict(zip(letters, [0]*9001))
  for c in s: counts[c] += 1
  counts = reversed(sorted([y for x, y in counts.iteritems() if y > 0]))
  return sum(map(lambda (x, y): x*y, zip(scores, counts)))

n = int(raw_input())
for _ in xrange(n):
  print 'Case #%d: %d' % (_+1, solve(raw_input()))
