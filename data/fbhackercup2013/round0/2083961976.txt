#include<stdio.h>
#include<string.h>
#include<ctype.h>
#include<algorithm>
using namespace std;
int n;
char s[111];
int dp[111][111];
int d(int l,int r){
	int&res=dp[l][r];
	if(res!=-1)return res;
	if(r-l==0)return res=1;
	if(r-l==1)return res=(s[l]!='(' && s[l]!=')');
	if(r-l==2){
		char L=s[l],R=s[r-1];
		if(L==':' && R=='(')return res=1;
		if(L==':' && R==')')return res=1;
	}
	if(s[l]=='(' && s[r-1]==')' && d(l+1,r-1))return res=1;
	for(int i=l+1; i<r; i++)
		if(d(l,i) && d(i,r))
			return res=1;
	return res=0;
}
int main(){
	int _;
	gets(s);
	sscanf(s,"%d",&_);
	for(int t=1; t<=_; t++){
		gets(s);
		n=strlen(s);
		memset(dp,-1,sizeof(dp));
		printf("Case #%d: %s\n",t,d(0,n)?"YES":"NO");
	}
	return 0;
}
