#define FOR(i, s, t) for(i = (s); i < (t); i++)
#define RFOR(i, s, t) for(i = (s)-1; i >= (t); i--)

char s[1000];
int cnt[26];

int main()
{
    #ifdef __FIO
    freopen("in.txt", "r", stdin);
    freopen("out.txt", "w", stdout);
    #endif
    int T, i0;
    scanf("%d", &T);
    gets(s);
    for(i0 = 1; i0 <= T; i0++){
        int n;
        int i;
        int ans;
        gets(s);
        n = strlen(s);
        memset(cnt, 0, sizeof cnt);
        for(i = 0; i < n; i++){
            if(s[i] >= 'a' && s[i] <= 'z')
                cnt[s[i]-'a']++;
            else if(s[i] >= 'A' && s[i] <= 'Z')
                cnt[s[i]-'A']++;
        }
        sort(cnt, cnt+26);
        ans = 0;
        for(i = 0; i < 26; i++)
            ans += cnt[i]*(i+1);
        printf("Case #%d: %d\n", i0, ans);
    }
    return 0;
}
