#include <algorithm>
#include <iostream>
#include <string>
#include <vector>

using namespace std;

int main() {
  int m; cin >> m; string dummy; getline(cin, dummy);
  for (int c = 1; c <= m; ++c) {
    string s; getline(cin, s);

    vector<int> count(26);
    for (int i = 0; i < s.length(); ++i) {
      if ('A' <= s[i] && s[i] <= 'Z') {
        count[s[i]-'A']++;
      } else if ('a' <= s[i] && s[i] <= 'z') {
        count[s[i]-'a']++;
      }
    }
    sort(count.begin(), count.end());
    int total = 0;
    for (int i = 0; i < count.size(); ++i) {
      total += count[i] * (i+1);
    }
    cout << "Case #" << c << ": " << total << endl;
  }
  return 0;
}
