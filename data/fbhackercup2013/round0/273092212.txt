#include <algorithm>
#include <stdio.h>
#include <string.h>
#include <ctype.h>

using namespace std;
int deal(char *s){
	int a[30] = {0};
	int len = strlen(s);
	for (int i = 0; i < len; i++) {
		char tmp = s[i];
		if (!isalpha(tmp)) 
			continue;
		if (tmp >= 'A' && tmp <= 'Z') {
			tmp = tmp - 'A' + 'a';
		}
		a[tmp - 'a']++;
	}
	sort(a, a + 26);
	int ans = 0;
	for (int i = 25; i >= 0; i--) {
		ans += a[i] * (i + 1);
	}
	return ans;
}
int main() {
	freopen("beautiful_stringstxt.txt", "r", stdin);
	freopen("out.txt", "w", stdout);
	int m, cases = 0;
	scanf("%d", &m);
	getchar();
	char str[550];
	while (m--) {
		memset(str, 0, sizeof(str));
		gets(str);
		printf("Case #%d: %d\n", ++cases, deal(str));
	}
	return 0;
}
