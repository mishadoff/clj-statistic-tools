#include <iostream>
#include <string>
#include <map>
#include <vector>
#include <algorithm>
using namespace std;

int main(void) {
	int m;
	cin >> m;
	string s;
	getline(cin,s);
	for(int ts=1; ts<=m; ts++) {
		cout << "Case #" << ts << ": ";
		getline(cin,s);
		map<int,int> cnt;
		for(int i=0; i<s.size(); i++) if(isalpha(s[i])) cnt[tolower(s[i])-'a']++;
		vector<int> cnts;
		for(int i=0; i<26; i++) cnts.push_back(cnt[i]);
		sort(cnts.begin(),cnts.end());
		reverse(cnts.begin(),cnts.end());
		int ret=0;
		for(int i=0; i<26; i++) ret+=cnts[i]*(26-i);
		cout << ret << endl;
	}
}
