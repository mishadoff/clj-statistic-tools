#include <iostream>
#include <vector>
#include <algorithm>
#include <string>
using namespace std;
int main(){
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);

	char tmp_str[600];
	
	int n;
	cin >> n;
	cin.getline(tmp_str, 600);
	
	vector<int> a(26);
	int b;
	for(int k = 0; k < n; k++) {
		b = 0;
		cin.getline(tmp_str, 600);

		string s = tmp_str;
		
		for(int i = 0; i < 26; i++)
			a[i] = 0;
		for(int i = 0; i < s.length(); i++) {
			if (s[i] >= 'A' && s[i] <= 'Z')
				s[i] += 'a' - 'A';
			if (s[i] >= 'a' && s[i] <= 'z')
				a[s[i] - 'a']++;
		}
		sort(a.begin(), a.end());
		for(int i = 0; i < 26; i++) {
			b += a[i] * (i + 1);
		}
		cout << "Case #" << k + 1 << ": " << b;
		if (k != n - 1)
			cout << endl;
	}
	return 0;
}
