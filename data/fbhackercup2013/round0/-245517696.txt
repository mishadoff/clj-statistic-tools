#include <cstdio>
#include <string>
#include <iostream>
#include <algorithm>
using namespace std;
int main()
{
	freopen("beautiful_stringstxt.txt","r",stdin);
	freopen("beautiful_outputtxt","w",stdout);
	int T;
	string tmp;
	getline(cin,tmp);
	sscanf(tmp.c_str(),"%d",&T);
	for(int kase=1;kase<=T;++kase)
	{
		getline(cin,tmp);
		int cnt[26]={0,};
		for (int i=0;i<tmp.size();++i) 
		{
			if (tmp[i]>='a' && tmp[i]<='z') cnt[tmp[i]-'a']++;
			else if (tmp[i]>='A' && tmp[i]<='Z') cnt[tmp[i]-'A']++;
		}
		sort(cnt,cnt+26);
		int ret = 0;
		for (int i=25;i>=0;--i)
			ret += cnt[i] * (i+1);
		printf("Case #%d: %d\n",kase,ret);
	}
	return 0;
}
