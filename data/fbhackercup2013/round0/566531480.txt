char arr[ MAX_LENGTH ] ;
int memo[ MAX_LENGTH ][ MAX_LENGTH ] , N ;

int dp( int i , int taken ) {
    if( i >= N ) {
        if( taken == 0 ) {
            return 1 ;
        }
        return 0 ;
    }
    if( taken < 0 ) {
        return 0 ;
    }
    int &ret = memo[ i ][ taken + 101 ] ;
    if( ret != -1 ) {
        return ret ;
    }
    int res ;
    res = 0 ;
    if( arr[ i ] == '(' || arr[ i ] == ')' ) {
        if( arr[ i ] == '(' ) {
            res |= dp( i + 1 , taken + 1 ) ;
        }
        if( arr[ i ] == ')' ) {
            res |= dp( i + 1 , taken - 1 ) ;
        }
        if( arr[ i ] == '(' && i - 1 >= 0 && arr[ i - 1 ] == ':' ) {
            res |= dp( i + 1 , taken ) ;
        }
        if( arr[ i ] == ')' && i - 1 >= 0 && arr[ i - 1 ] == ':' ) {
            res |= dp( i + 1 , taken ) ;
        }
    }
    else {
        res |= dp( i + 1 , taken ) ;
    }
    return ret = res ;
}
